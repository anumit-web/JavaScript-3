Array CheatSheet

(['a', 'b'].concat(['c'])       // ['a', 'b','c']
(['a', 'b'].join(['-'])       // ['a-b']
(['a', 'b','c'].slice(1)       // ['b','c']
(['a', 'b','c'].indexOf('b')       // 1
(['a', 'b','b'].lastIndexOf('b')       // 2
([1, 2, 3].map(x => x*2)       // [2, 4, 6]
([1, 4, 9].map(num => Math.sqrt(num))  // [1, 2, 3]
([2, 15, 3].sort()       // [15, 3, 2]
([1, 2, 3].reverse()       // [3, 2, 1]
