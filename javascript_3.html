Array CheatSheet

(['a', 'b'].concat(['c'])       // ['a', 'b','c']
(['a', 'b'].join(['-'])       // ['a-b']
(['a', 'b','c'].slice(1)       // ['b','c']
(['a', 'b','c'].indexOf('b')       // 1
(['a', 'b','b'].lastIndexOf('b')       // 2
([1, 2, 3].map(x => x*2)       // [2, 4, 6]
([1, 4, 9].map(num => Math.sqrt(num))  // [1, 2, 3]
([2, 15, 3].sort()       // [15, 3, 2]
([1, 2, 3].reverse()       // [3, 2, 1]
([1, 2, 3].reduce((x,y) => (x*y))       // [3, 2, 1]
([1, 2, 3].length       // 3
([1, 2, 3].every(x => x < 10)  // true
([1, 2, 3].some(x => x < 2)  // true

const arr=[1, 2, 3, 4]

x=arr.shift()  // [2, 3, 4]]
x=arr.unshift(9)  // [9, 1, 2, 3, 4]]
x=arr.pop()  // arr = [1, 2] , x = 3
x=arr.push(55)  // [1, 2, 3, 55]]
